621  static int
622 py3_runtime_link_init(char *libname, int verbose)
623 {
624  int i;
625  PYTHON_PROC *ucs_from_string = (PYTHON_PROC *)&py3_PyUnicode_FromString;
626  PYTHON_PROC *ucs_decode = (PYTHON_PROC *)&py3_PyUnicode_Decode;

628  (PYTHON_PROC *)&py3_PyUnicode_AsEncodedString;

631  /* Can't have Python and Python3 loaded at the same time.
632  * It cause a crash, because RTLD_GLOBAL is needed for
633  * standard C extension libraries of one or both python versions. */
634  if (python_loaded())
635  {
636  if (verbose)
637  emsg(_("E837: This Vim cannot execute :py3 after using :python"));
638  return FAIL;
639  }
640 # endif
641 
642  if (hinstPy3 != 0)
643  return OK;
644  hinstPy3 = load_dll(libname);
645 
646  if (!hinstPy3)
647  {
648  if (verbose)
649  semsg(_(e_loadlib), libname);
650  return FAIL;
651  }
652 
653  for (i = 0; py3_funcname_table[i].ptr; ++i)
654  {
655  if ((*py3_funcname_table[i].ptr = symbol_from_dll(hinstPy3,
656  py3_funcname_table[i].name)) == NULL)
657  {
658  close_dll(hinstPy3);
659  hinstPy3 = 0;
660  if (verbose)
661  semsg(_(e_loadfunc), py3_funcname_table[i].name);
662  return FAIL;
663  }
664  }
665 
666  /* Load unicode functions separately as only the ucs2 or the ucs4 functions
667  * will be present in the library. */
668 # if PY_VERSION_HEX >= 0x030300f0
669  *ucs_from_string = symbol_from_dll(hinstPy3, "PyUnicode_FromString");
670  *ucs_decode = symbol_from_dll(hinstPy3, "PyUnicode_Decode");
671  *ucs_as_encoded_string = symbol_from_dll(hinstPy3,
672  "PyUnicode_AsEncodedString");

675  *ucs_decode = symbol_from_dll(hinstPy3,
676  "PyUnicodeUCS2_Decode");

678  "PyUnicodeUCS2_AsEncodedString");

682  *ucs_from_string = symbol_from_dll(hinstPy3,

684  *ucs_decode = symbol_from_dll(hinstPy3,
685  "PyUnicodeUCS4_Decode");
