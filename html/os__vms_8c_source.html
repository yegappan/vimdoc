365 /*
366  * vms_wproc() is called for each matching filename by decc$to_vms().
367  * We want to save each match for later retrieval.
368  *
369  * Returns: 1 - continue finding matches
370  * 0 - stop trying to find any further matches
371  */
372  static int
373 vms_wproc(char *name, int val)
374 {
375  int i;
376  static int vms_match_alloced = 0;
377 
378  if (val == DECC$K_FOREIGN ) /* foreign non VMS files are not counting */
379  return 1;
380 
381  /* accept all DECC$K_FILE and DECC$K_DIRECTORY */
382  if (vms_match_num == 0) {
383  /* first time through, setup some things */
384  if (NULL == vms_fmatch) {
385  vms_fmatch = ALLOC_MULT(char *, EXPL_ALLOC_INC);
386  if (!vms_fmatch)
387  return 0;
388  vms_match_alloced = EXPL_ALLOC_INC;
389  vms_match_free = EXPL_ALLOC_INC;
390  }
391  else {
392  /* re-use existing space */
393  vms_match_free = vms_match_alloced;
394  }
395  }
396 
397  /* make matches look uniform */
398  vms_remove_version(name);
399  name=vms_tolower(name);
400 
401  /* if name already exists, don't add it */
402  for (i = 0; i<vms_match_num; i++) {
403  if (0 == STRCMP((char_u *)name,vms_fmatch[i]))
404  return 1;
405  }
406  if (--vms_match_free == 0) {
407  /* add more space to store matches */
408  vms_match_alloced += EXPL_ALLOC_INC;
409  vms_fmatch = vim_realloc(vms_fmatch,
410  sizeof(char **) * vms_match_alloced);
411  if (!vms_fmatch)
412  return 0;
413  vms_match_free = EXPL_ALLOC_INC;
414  }
415  vms_fmatch[vms_match_num] = vim_strsave((char_u *)name);
416 
417  ++vms_match_num;
418  return 1;
419 }

505  }

534 /*
535  * attempt to translate a mixed unix-vms file specification to pure vms
536  */
537  static void
538 vms_unix_mixed_filespec(char *in, char *out)
539 {
540  char *lastcolon;
541  char *end_of_dir;
542  char ch;
543  int len;
544  char *out_str=out;
545 
546  /* copy vms filename portion up to last colon
547  * (node and/or disk)
548  */
549  lastcolon = strrchr(in, ':'); /* find last colon */
550  if (lastcolon != NULL) {
551  len = lastcolon - in + 1;
552  strncpy(out, in, len);
553  out += len;
554  in += len;
555  }
556 
557  end_of_dir = NULL; /* default: no directory */
558 
559  /* start of directory portion */
560  ch = *in;
561  if ((ch == '[') || (ch == '/') || (ch == '<')) { /* start of directory(s) ? */
562  ch = '[';
563  SKIP_FOLLOWING_SLASHES(in);
564  } else if (EQN(in, "../", 3)) { /* Unix parent directory? */
565  *out++ = '[';
566  *out++ = '-';
567  end_of_dir = out;
568  ch = '.';
569  in += 2;
570  SKIP_FOLLOWING_SLASHES(in);
571  } else { /* not a special character */
572  while (EQN(in, "./", 2)) { /* Ignore Unix "current dir" */
573  in += 2;
574  SKIP_FOLLOWING_SLASHES(in);
575  }
576  if (strchr(in, '/') == NULL) { /* any more Unix directories ? */
577  strcpy(out, in); /* No - get rest of the spec */
578  return;
579  } else {
580  *out++ = '['; /* Yes, denote a Vms subdirectory */
581  ch = '.';
582  --in;
583  }
584  }
585 
586  /* if we get here, there is a directory part of the filename */
587 
588  /* initialize output file spec */
589  *out++ = ch;
590  ++in;
591 
592  while (*in != '\0') {
593  ch = *in;
594  if ((ch == ']') || (ch == '/') || (ch == '>') ) { /* end of (sub)directory ? */
595  end_of_dir = out;
596  ch = '.';
597  SKIP_FOLLOWING_SLASHES(in);
598  }
599  else if (EQN(in, "../", 3)) { /* Unix parent directory? */
600  *out++ = '-';
601  end_of_dir = out;
602  ch = '.';
603  in += 2;
604  SKIP_FOLLOWING_SLASHES(in);
605  }
606  else {
607  while (EQN(in, "./", 2)) { /* Ignore Unix "current dir" */
608  end_of_dir = out;
609  in += 2;
610  SKIP_FOLLOWING_SLASHES(in);
611  ch = *in;
612  }
613  }
614 
615  /* Place next character into output file spec */
616  *out++ = ch;
617  ++in;
618  }
619 
620  *out = '\0'; /* Terminate output file spec */
621 
622  if (end_of_dir != NULL) /* Terminate directory portion */
623  *end_of_dir = ']';
624 }

773

795  return 0; /* error, incl. time_diff < 0 (i.e. time-out) */
796 
797  /* otherwise wait some more */
798  }
