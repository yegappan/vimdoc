395 /*
396  * Initialize the crypt method using "password" as the encryption key and
397  * "salt[salt_len]" as the salt.
398  */
399  static void
400 bf_key_init(
401  bf_state_T *bfs,
402  char_u *password,
403  char_u *salt,
404  int salt_len)
405 {
406  int i, j, keypos = 0;
407  unsigned u;
408  UINT32_T val, data_l, data_r;
409  char_u *key;
410  int keylen;
411 
412  /* Process the key 1001 times.
413  * See http://en.wikipedia.org/wiki/Key_strengthening. */
414  key = sha256_key(password, salt, salt_len);
415  for (i = 0; i < 1000; i++)
416  key = sha256_key(key, salt, salt_len);
417 
418  /* Convert the key from 64 hex chars to 32 binary chars. */
419  keylen = (int)STRLEN(key) / 2;
420  if (keylen == 0)
421  {
422  iemsg(_("E831: bf_key_init() called with empty password"));
423  return;
424  }
425  for (i = 0; i < keylen; i++)
426  {
427  sscanf((char *)&key[i * 2], "%2x", &u);
428  key[i] = u;
429  }
430 
431  /* Use "key" to initialize the P-array ("pax") and S-boxes ("sbx") of
432  * Blowfish. */
433  mch_memmove(bfs->sbx, sbx_init, 4 * 4 * 256);
434 
435  for (i = 0; i < 18; ++i)
436  {
437  val = 0;
438  for (j = 0; j < 4; ++j)
439  val = (val << 8) | key[keypos++ % keylen];
440  bfs->pax[i] = pax_init[i] ^ val;
441  }
442 
443  data_l = data_r = 0;
444  for (i = 0; i < 18; i += 2)
445  {
446  bf_e_block(bfs, &data_l, &data_r);
447  bfs->pax[i + 0] = data_l;
448  bfs->pax[i + 1] = data_r;
449  }
450 
451  for (i = 0; i < 4; ++i)
452  {
453  for (j = 0; j < 256; j += 2)
454  {
455  bf_e_block(bfs, &data_l, &data_r);
456  bfs->sbx[i][j + 0] = data_l;
457  bfs->sbx[i][j + 1] = data_r;
458  }
459  }
