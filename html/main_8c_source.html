1984 command_line_scan(mparm_T *parmp)
1985 {
1986  int argc = parmp->argc;
1987  char **argv = parmp->argv;
1988  int argv_idx; /* index in argv[n][] */

1990  int want_argument; /* option argument with argument */
1991  int c;
1992  char_u *p = NULL;
1993  long n;
1994 
1995  --argc;
1996  ++argv;
1997  argv_idx = 1; /* active option letter is argv[0][argv_idx] */
1998  while (argc > 0)
1999  {
2000  /*
2001  * "+" or "+{number}" or "+/{pat}" or "+{command}" argument.
2002  */
2003  if (argv[0][0] == '+' && !had_minmin)
2004  {
2005  if (parmp->n_commands >= MAX_ARG_CMDS)
2006  mainerr(ME_EXTRA_CMD, NULL);
2007  argv_idx = -1; /* skip to next argument */
2008  if (argv[0][1] == NUL)
2009  parmp->commands[parmp->n_commands++] = (char_u *)"$";
2010  else
2011  parmp->commands[parmp->n_commands++] = (char_u *)&(argv[0][1]);
2012  }

2015  * Optional argument.
2016  */
2017  else if (argv[0][0] == '-' && !had_minmin)
2018  {
2019  want_argument = FALSE;
2020  c = argv[0][argv_idx++];
2021 #ifdef VMS

2023  * VMS only uses upper case command lines. Interpret "-X" as "-x"
2024  * and "-/X" as "-X".
2025  */
2026  if (c == '/')
2027  {
2028  c = argv[0][argv_idx++];
2029  c = TOUPPER_ASC(c);
2030  }
2031  else
2032  c = TOLOWER_ASC(c);

2035  {

2037  /* "ex -" silent mode */
2038  if (exmode_active)

2042  if (parmp->edit_type != EDIT_NONE)
