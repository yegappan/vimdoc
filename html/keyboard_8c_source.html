7 void vterm_keyboard_unichar(VTerm *vt, uint32_t c, VTermModifier mod)
8 {
9  int needs_CSIu;
10 
11  // The shift modifier is never important for Unicode characters
12  // apart from Space
13  if(c != ' ')
14  mod &= ~VTERM_MOD_SHIFT;
15 
16  if(mod == 0) {
17  // Normal text - ignore just shift
18  char str[6];
19  int seqlen = fill_utf8(c, str);
20  vterm_push_output_bytes(vt, str, seqlen);
21  return;
22  }
23 
24  switch(c) {
25  // Special Ctrl- letters that can't be represented elsewise
26  case 'i': case 'j': case 'm': case '[':
27  needs_CSIu = 1;
28  break;
29  // Ctrl-\ ] ^ _ don't need CSUu
30  case '\\': case ']': case '^': case '_':
31  needs_CSIu = 0;
32  break;
33  // Shift-space needs CSIu
34  case ' ':
35  needs_CSIu = !!(mod & VTERM_MOD_SHIFT);
36  break;
37  // All other characters needs CSIu except for letters a-z
38  default:
39  needs_CSIu = (c < 'a' || c > 'z');
40  }
41 
42  // ALT we can just prefix with ESC; anything else requires CSI u
43  if(needs_CSIu && (mod & ~VTERM_MOD_ALT)) {
44  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "%d;%du", c, mod+1);
45  return;
46  }
47 
48  if(mod & VTERM_MOD_CTRL)
49  c &= 0x1f;
50 
51  vterm_push_output_sprintf(vt, "%s%c", mod & VTERM_MOD_ALT ? ESC_S : "", c);
52 }

129 void vterm_keyboard_key(VTerm *vt, VTermKey key, VTermModifier mod)
130 {
131  keycodes_s k;
132 
133  if(key == VTERM_KEY_NONE)
134  return;
135 
136  if(key < VTERM_KEY_FUNCTION_0) {
137  if(key >= sizeof(keycodes)/sizeof(keycodes[0]))
138  return;
139  k = keycodes[key];
140  }
141  else if(key >= VTERM_KEY_FUNCTION_0 && key <= VTERM_KEY_FUNCTION_MAX) {
142  if((key - VTERM_KEY_FUNCTION_0) >= sizeof(keycodes_fn)/sizeof(keycodes_fn[0]))
143  return;
144  k = keycodes_fn[key - VTERM_KEY_FUNCTION_0];
145  }
146  else if(key >= VTERM_KEY_KP_0) {
147  if((key - VTERM_KEY_KP_0) >= sizeof(keycodes_kp)/sizeof(keycodes_kp[0]))
148  return;
149  k = keycodes_kp[key - VTERM_KEY_KP_0];
150  }
151 
152  switch(k.type) {
153  case KEYCODE_NONE:
154  break;
155 
156  case KEYCODE_TAB:
157  // Shift-Tab is CSI Z but plain Tab is 0x09
158  if(mod == VTERM_MOD_SHIFT)
159  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "Z");
160  else if(mod & VTERM_MOD_SHIFT)
161  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "1;%dZ", mod+1);
162  else
163  goto case_LITERAL;
164  break;
165 
166  case KEYCODE_ENTER:
167  // Enter is CRLF in newline mode, but just LF in linefeed
168  if(vt->state->mode.newline)
169  vterm_push_output_sprintf(vt, "\r\n");
170  else
171  goto case_LITERAL;
172  break;
173 
174  case KEYCODE_LITERAL: case_LITERAL:
175  if(mod & (VTERM_MOD_SHIFT|VTERM_MOD_CTRL))
176  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "%d;%du", k.literal, mod+1);
177  else
178  vterm_push_output_sprintf(vt, mod & VTERM_MOD_ALT ? ESC_S "%c" : "%c", k.literal);
179  break;
180 
181  case KEYCODE_SS3: case_SS3:
182  if(mod == 0)
183  vterm_push_output_sprintf_ctrl(vt, C1_SS3, "%c", k.literal);
184  else
185  goto case_CSI;
186  break;
187 
188  case KEYCODE_CSI: case_CSI:
189  if(mod == 0)
190  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "%c", k.literal);
191  else
192  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "1;%d%c", mod + 1, k.literal);
193  break;
194 
195  case KEYCODE_CSINUM:
196  if(mod == 0)
197  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "%d%c", k.csinum, k.literal);
198  else
199  vterm_push_output_sprintf_ctrl(vt, C1_CSI, "%d;%d%c", k.csinum, mod + 1, k.literal);
200  break;
201 
202  case KEYCODE_CSI_CURSOR:
203  if(vt->state->mode.cursor)
204  goto case_SS3;
205  else
206  goto case_CSI;
207 
208  case KEYCODE_KEYPAD:
209  if(vt->state->mode.keypad) {
210  k.literal = k.csinum;
211  goto case_SS3;
212  }
213  else
214  goto case_LITERAL;
215  }
