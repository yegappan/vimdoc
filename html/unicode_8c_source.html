10 /*
11  * This is an implementation of wcwidth() and wcswidth() (defined in
12  * IEEE Std 1002.1-2001) for Unicode.
13  *
14  * http://www.opengroup.org/onlinepubs/007904975/functions/wcwidth.html
15  * http://www.opengroup.org/onlinepubs/007904975/functions/wcswidth.html
16  *
17  * In fixed-width output devices, Latin characters all occupy a single
18  * "cell" position of equal width, whereas ideographic CJK characters
19  * occupy two such cells. Interoperability between terminal-line
20  * applications and (teletype-style) character terminals using the
21  * UTF-8 encoding requires agreement on which character should advance
22  * the cursor by how many cell positions. No established formal
23  * standards exist at present on which Unicode character shall occupy
24  * how many cell positions on character terminals. These routines are
25  * a first attempt of defining such behavior based on simple rules
26  * applied to data provided by the Unicode Consortium.
27  *
28  * For some graphical characters, the Unicode standard explicitly
29  * defines a character-cell width via the definition of the East Asian
30  * FullWidth (F), Wide (W), Half-width (H), and Narrow (Na) classes.
31  * In all these cases, there is no ambiguity about which width a
32  * terminal shall use. For characters in the East Asian Ambiguous (A)
33  * class, the width choice depends purely on a preference of backward
34  * compatibility with either historic CJK or Western practice.
35  * Choosing single-width for these characters is easy to justify as
36  * the appropriate long-term solution, as the CJK practice of
37  * displaying these characters as double-width comes from historic
38  * implementation simplicity (8-bit encoded characters were displayed
39  * single-width and 16-bit ones double-width, even for Greek,
40  * Cyrillic, etc.) and not any typographic considerations.
41  *
42  * Much less clear is the choice of width for the Not East Asian
43  * (Neutral) class. Existing practice does not dictate a width for any
44  * of these characters. It would nevertheless make sense
45  * typographically to allocate two character cells to characters such
46  * as for instance EM SPACE or VOLUME INTEGRAL, which cannot be
47  * represented adequately with a single-width glyph. The following
48  * routines at present merely assign a single-cell width to all
49  * neutral characters, in the interest of simplicity. This is not
50  * entirely satisfactory and should be reconsidered before
51  * establishing a formal standard in this area. At the moment, the
52  * decision which Not East Asian (Neutral) characters should be
53  * represented by double-width glyphs cannot yet be answered by
54  * applying a simple rule from the Unicode database content. Setting
55  * up a proper standard for the behavior of UTF-8 character terminals
56  * will require a careful analysis not only of each Unicode character,
57  * but also of each presentation form, something the author of these
58  * routines has avoided to do so far.
59  *
60  * http://www.unicode.org/unicode/reports/tr11/
61  *
62  * Markus Kuhn -- 2007-05-26 (Unicode 5.0)
63  *
64  * Permission to use, copy, modify, and distribute this software
65  * for any purpose and without fee is hereby granted. The author
66  * disclaims all warranties with regard to this software.
67  *
68  * Latest version: http://www.cl.cam.ac.uk/~mgk25/ucs/wcwidth.c
69  */

365 // auxiliary function for binary search in interval table
366 static int bisearch(uint32_t ucs, const struct interval *table, int max) {
367  int min = 0;
368  int mid;
369 
370  if ((int)ucs < table[0].first || (int)ucs > table[max].last)
371  return 0;
372  while (max >= min) {
373  mid = (min + max) / 2;
374  if ((int)ucs > table[mid].last)
375  min = mid + 1;
376  else if ((int)ucs < table[mid].first)
377  max = mid - 1;
378  else
379  return 1;
380  }
381 
382  return 0;
383 }

385 /* The following two functions define the column width of an ISO 10646
386  * character as follows:
387  *
388  * - The null character (U+0000) has a column width of 0.
389  *
390  * - Other C0/C1 control characters and DEL will lead to a return
391  * value of -1.
392  *
393  * - Non-spacing and enclosing combining characters (general
394  * category code Mn or Me in the Unicode database) have a
395  * column width of 0.
396  *
397  * - SOFT HYPHEN (U+00AD) has a column width of 1.
398  *
399  * - Other format characters (general category code Cf in the Unicode
400  * database) and ZERO WIDTH SPACE (U+200B) have a column width of 0.
401  *
402  * - Hangul Jamo medial vowels and final consonants (U+1160-U+11FF)
403  * have a column width of 0.
404  *
405  * - Spacing characters in the East Asian Wide (W) or East Asian
406  * Full-width (F) category as defined in Unicode Technical
407  * Report #11 have a column width of 2.
408  *
409  * - All remaining characters (including all printable
410  * ISO 8859-1 and WGL4 characters, Unicode control characters,
411  * etc.) have a column width of 1.
412  *
413  * This implementation assumes that uint32_t characters are encoded
414  * in ISO 10646.
415  */

423 static int mk_wcwidth(uint32_t ucs)
424 {
425  // test for 8-bit control characters
426  if (ucs == 0)
427  return 0;
428  if (ucs < 32 || (ucs >= 0x7f && ucs < 0xa0))
429  return -1;
430 
431  // binary search in table of non-spacing characters
432  if (bisearch(ucs, combining,
433  sizeof(combining) / sizeof(struct interval) - 1))
434  return 0;
435 
436  // if we arrive here, ucs is not a combining or C0/C1 control character
437 
438  return 1 +
439  (ucs >= 0x1100 &&
440  (ucs <= 0x115f || // Hangul Jamo init. consonants
441  ucs == 0x2329 || ucs == 0x232a ||
442  (ucs >= 0x2e80 && ucs <= 0xa4cf &&
443  ucs != 0x303f) || // CJK ... Yi
444  (ucs >= 0xac00 && ucs <= 0xd7a3) || // Hangul Syllables
445  (ucs >= 0xf900 && ucs <= 0xfaff) || // CJK Compatibility Ideographs
446  (ucs >= 0xfe10 && ucs <= 0xfe19) || // Vertical forms
447  (ucs >= 0xfe30 && ucs <= 0xfe6f) || // CJK Compatibility Forms
448  (ucs >= 0xff00 && ucs <= 0xff60) || // Fullwidth Forms
449  (ucs >= 0xffe0 && ucs <= 0xffe6) ||
450  (ucs >= 0x20000 && ucs <= 0x2fffd) ||
451  (ucs >= 0x30000 && ucs <= 0x3fffd)));
452 }
