536 /*
537  * Translate Vim's mnemonic tagging to GTK+ style and convert to UTF-8
538  * if necessary. The caller must vim_free() the returned string.
539  *
540  * Input Output
541  * _ __
542  * && &
543  * & _ stripped if use_mnemonic == FALSE
544  * <Tab> end of menu label text
545  */
546  static char_u *
547 translate_mnemonic_tag(char_u *name, int use_mnemonic)
548 {
549  char_u *buf;
550  char_u *psrc;
551  char_u *pdest;
552  int n_underscores = 0;
553 
554  name = CONVERT_TO_UTF8(name);
555  if (name == NULL)
556  return NULL;
557 
558  for (psrc = name; *psrc != NUL && *psrc != TAB; ++psrc)
559  if (*psrc == '_')
560  ++n_underscores;
561 
562  buf = alloc(psrc - name + n_underscores + 1);
563  if (buf != NULL)
564  {
565  pdest = buf;
566  for (psrc = name; *psrc != NUL && *psrc != TAB; ++psrc)
567  {
568  if (*psrc == '_')
569  {
570  *pdest++ = '_';
571  *pdest++ = '_';
572  }
573  else if (*psrc != '&')
574  {
575  *pdest++ = *psrc;
576  }
577  else if (*(psrc + 1) == '&')
578  {
579  *pdest++ = *psrc++;
580  }
581  else if (use_mnemonic)
582  {
583  *pdest++ = '_';
584  }
585  }
586  *pdest = NUL;
587  }
588 
589  CONVERT_TO_UTF8_FREE(name);
590  return buf;
591 }

1167 /*
1168  * Put up a file requester.
1169  * Returns the selected name in allocated memory, or NULL for Cancel.
1170  * saving, select file to write
1171  * title title for the window
1172  * dflt default name
1173  * ext not used (extension added)
1174  * initdir initial directory, NULL for current dir
1175  * filter not used (file name filter)
1176  */
1177  char_u *
1178 gui_mch_browse(int saving UNUSED,
1179  char_u *title,
1180  char_u *dflt,
1181  char_u *ext UNUSED,
1182  char_u *initdir,
1183  char_u *filter)
1184 {
1185 #ifdef USE_FILE_CHOOSER
1186  GtkWidget *fc;
1187 #endif
1188  char_u dirbuf[MAXPATHL];
1189  guint log_handler;
1190  const gchar *domain = "Gtk";
1191 
1192  title = CONVERT_TO_UTF8(title);
1193 
1194  /* GTK has a bug, it only works with an absolute path. */
1195  if (initdir == NULL || *initdir == NUL)
1196  mch_dirname(dirbuf, MAXPATHL);
1197  else if (vim_FullName(initdir, dirbuf, MAXPATHL - 2, FALSE) == FAIL)
1198  dirbuf[0] = NUL;
1199  /* Always need a trailing slash for a directory. */
1200  add_pathsep(dirbuf);
1201 
1202  /* If our pointer is currently hidden, then we should show it. */
1203  gui_mch_mousehide(FALSE);
1204 
1205  /* Hack: The GTK file dialog warns when it can't access a new file, this
1206  * makes it shut up. http://bugzilla.gnome.org/show_bug.cgi?id=664587 */
1207  log_handler = g_log_set_handler(domain, G_LOG_LEVEL_WARNING,
1208  recent_func_log_func, NULL);
1209 
1210 #ifdef USE_FILE_CHOOSER
1211  /* We create the dialog each time, so that the button text can be "Open"
1212  * or "Save" according to the action. */
1213  fc = gtk_file_chooser_dialog_new((const gchar *)title,
1214  GTK_WINDOW(gui.mainwin),
1215  saving ? GTK_FILE_CHOOSER_ACTION_SAVE
1216  : GTK_FILE_CHOOSER_ACTION_OPEN,
1217 # if GTK_CHECK_VERSION(3,10,0)
1218  _("_Cancel"), GTK_RESPONSE_CANCEL,
1219  saving ? _("_Save") : _("_Open"), GTK_RESPONSE_ACCEPT,
1220 # else
1221  GTK_STOCK_CANCEL, GTK_RESPONSE_CANCEL,
1222  saving ? GTK_STOCK_SAVE : GTK_STOCK_OPEN, GTK_RESPONSE_ACCEPT,
1223 # endif
1224  NULL);
1225  gtk_file_chooser_set_current_folder(GTK_FILE_CHOOSER(fc),
1226  (const gchar *)dirbuf);
1227 
1228  if (filter != NULL && *filter != NUL)
1229  {
1230  int i = 0;
1231  char_u *patt;
1232  char_u *p = filter;
1233  GtkFileFilter *gfilter;
1234 
1235  gfilter = gtk_file_filter_new();
1236  patt = alloc(STRLEN(filter));
1237  while (p != NULL && *p != NUL)
1238  {
1239  if (*p == '\n' || *p == ';' || *p == '\t')
1240  {
1241  STRNCPY(patt, filter, i);
1242  patt[i] = '\0';
1243  if (*p == '\t')
1244  gtk_file_filter_set_name(gfilter, (gchar *)patt);
1245  else
1246  {
1247  gtk_file_filter_add_pattern(gfilter, (gchar *)patt);
1248  if (*p == '\n')
1249  {
1250  gtk_file_chooser_add_filter((GtkFileChooser *)fc,
1251  gfilter);
1252  if (*(p + 1) != NUL)
1253  gfilter = gtk_file_filter_new();
1254  }
1255  }
1256  filter = ++p;
1257  i = 0;
1258  }
1259  else
1260  {
1261  p++;
1262  i++;
1263  }
1264  }
